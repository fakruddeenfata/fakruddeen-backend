# Yi amfani da ingantaccen hoton Python mai sauki
FROM python:3.11-slim

# Saita gurbin aiki a cikin container
WORKDIR /backend

# Sanya kayan aikin Linux da ake bukata don ginin libraries
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    && rm -rf /var/lib/apt/lists/*

# Kwafi fayil din requirements da farko don yin caching na libraries
COPY requirements.txt .

# Shigar da duka libraries da application din ke bukata
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir -r requirements.txt

# Maimakon zuba komai a waje, za mu samar da folder ta 'app' a cikin container
# mu zuba fayilolin a ciki domin 'from app.core' ta yi aiki lafiya
RUN mkdir -p app
COPY . ./app/

# Bude kofa ta 8000 domin duniya ta iya shiga
EXPOSE 8000

# Saita PYTHONPATH domin Python ya rika duba asalin '/backend'
ENV PYTHONPATH=/backend

# Umarnin da zai kunna uvicorn ta hanyar kiran 'app.main:app'
CMD ["python", "-m", "uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8000", "--workers", "4"]
